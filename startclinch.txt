    var input_query = document.getElementById('query');
    var resultsContainer = document.getElementById('autocomplete-results');
    var searchIcon = document.getElementById('autofocus-trigger');
    function debounce(func, wait) {
      let timeout;
      return function (...args) {
        clearTimeout(timeout);
        timeout = setTimeout(() => {
          func.apply(this, args);
        }, wait);
      };
    }
    var debounceInputHandler = debounce(() => {
      var inputValue = input_query.value.toLowerCase();
      fetch(`/search/autocomplete?q=${inputValue}`)
        .then(response => response.json())
        .then(data => {
          var filteredSuggestions = data;
          renderSuggestions(filteredSuggestions);
        })
        .catch(error => {
        //   console.error('Error fetching suggestions:', error);
        });
    }, 300);
    input_query.addEventListener('input', debounceInputHandler);
    function renderSuggestions(suggestions) {
      resultsContainer.innerHTML = '';
      suggestions.forEach(suggestion => {
        var suggestionElement = document.createElement('div');
        suggestionElement.classList.add('autocomplete-suggestion'); 
        suggestionElement.innerHTML = `
        <div class="suggestion">
            <div class="singer-info">
                <img src="${suggestion.profile_pic}" alt="${suggestion.professional_name}" class="singer-image">
                <div class="singer-details">
                    <span class="singer-name">${suggestion.professional_name}</span>
                    <p class="singer-description">${suggestion.category_name}</p>
                </div>
            </div>
        </div>
        `;
        suggestionElement.addEventListener('click', () => {
            input_query.value = suggestion.professional_name;
            window.location.href = `/${suggestion.slug}`;
        });
        resultsContainer.appendChild(suggestionElement);
      });
    }

    
    document.addEventListener('click', event => {
        if (!resultsContainer.contains(event.target) && event.target !== input_query) {
            resultsContainer.innerHTML = '';
        }
    });

            levenshtein: function(s1, s2) {
                var longer = s1;
                var shorter = s2;
                if (s1.length < s2.length) {
                    longer = s2;
                    shorter = s1;
                }
                var longerLength = longer.length;
                if (longerLength == 0) {
                    return 1.0;
                }
                return (longerLength - this.editDistance(longer, shorter)) / parseFloat(longerLength);
            },
            editDistance: function(s1, s2) {
                s1 = s1.toLowerCase();
                s2 = s2.toLowerCase();
    
                var costs = new Array();
                for (var i = 0; i <= s1.length; i++) {
                    var lastValue = i;
                    for (var j = 0; j <= s2.length; j++) {
                        if (i == 0)
                            costs[j] = j;
                        else {
                            if (j > 0) {
                                var newValue = costs[j - 1];
                                if (s1.charAt(i - 1) != s2.charAt(j - 1))
                                    newValue = Math.min(Math.min(newValue, lastValue),
                                        costs[j]) + 1;
                                costs[j - 1] = lastValue;
                                lastValue = newValue;
                            }
                        }
                    }
                    if (i > 0)
                        costs[s2.length] = lastValue;
                }
                return costs[s2.length];
            },

function showSlide(n) {

    slides[i].style.display = "none";
    }

    slides[n].style.display = "block";

    if (n === 0) {
    prevBtn.classList.add("hidden");
    } else {
    prevBtn.classList.remove("hidden");
    }

    if (n === slides.length - 1) {
    nextBtn.classList.add("hidden");
    } else {
    nextBtn.classList.remove("hidden");
    }
}

    console.log(currentSlide);
    if (currentSlide > 0) {
    currentSlide--;
    showSlide(currentSlide);
    }
}

function nextSlide() {
    console.log(currentSlide);

    if (currentSlide < slides.length - 1) {
    currentSlide++;
    showSlide(currentSlide);
    }
}
